    <script>
        // --- UI ELEMENTS ---
        const localVideo = document.getElementById('localVideo');
        const remoteVideo = document.getElementById('remoteVideo');
        const statusTextContainer = document.getElementById('statusText');
        const statusMsgContent = document.getElementById('statusMsgContent');
        const callTimerDisplay = document.getElementById('callTimer');
        
        const btnToggleMic = document.getElementById('btnToggleMic');
        const btnToggleCam = document.getElementById('btnToggleCam');
        const btnHangup = document.getElementById('btnHangup');

        // --- STATE VARIABLES ---
        let isMicOn = true;
        let isCamOn = true;
        let timerInterval = null;
        let totalSeconds = 0;

        let localStream = null;
        let peerConnection = null;
        let pendingCandidates = [];

        // Cấu hình STUN Server
        const config = {
            iceServers: [
                { urls: 'stun:stun.l.google.com:19302' },
                { urls: 'stun:stun1.l.google.com:19302' }
            ],
            iceCandidatePoolSize: 10
        };

        // ================= UI FUNCTIONS =================

        function startTimer() {
            stopTimer();
            totalSeconds = 0;
            callTimerDisplay.style.display = "block";
            callTimerDisplay.textContent = "00:00";
            timerInterval = setInterval(() => {
                totalSeconds++;
                const m = Math.floor(totalSeconds / 60).toString().padStart(2, '0');
                const s = (totalSeconds % 60).toString().padStart(2, '0');
                callTimerDisplay.textContent = `${m}:${s}`;
            }, 1000);
        }

        function stopTimer() {
            if (timerInterval) clearInterval(timerInterval);
            callTimerDisplay.style.display = "none";
        }

        function updateStatus(state) {
            const messages = {
                'new': 'Đang khởi tạo...',
                'checking': 'Kiểm tra mạng...',
                'connecting': 'Đang kết nối...',
                'connected': 'Đã kết nối',
                'disconnected': 'Mất kết nối',
                'failed': 'Kết nối thất bại',
                'closed': 'Cuộc gọi kết thúc'
            };
            
            statusMsgContent.textContent = messages[state] || state;
            statusTextContainer.classList.add('visible');
            
            if (state === 'connected') {
                statusTextContainer.classList.add('connected');
                startTimer();
            } else {
                statusTextContainer.classList.remove('connected');
                stopTimer();
            }
        }

        function updateRemoteState(type, state) {
            // Xử lý chuỗi JSON thừa ngoặc kép nếu có
            const cleanState = state.replace(/"/g, ''); 
            
            if (type === 'mic') {
                document.getElementById('remoteMuteIcon').style.display = (cleanState === 'off') ? 'flex' : 'none';
            } else if (type === 'cam') {
                document.getElementById('remoteCamIcon').style.display = (cleanState === 'off') ? 'flex' : 'none';
            }
        }

        // ================= BUTTON EVENTS =================

        btnToggleMic.addEventListener('click', () => {
            isMicOn = !isMicOn;
            const icon = btnToggleMic.querySelector('i');
            if (isMicOn) {
                icon.className = 'fa-solid fa-microphone';
                btnToggleMic.classList.remove('active-off');
            } else {
                icon.className = 'fa-solid fa-microphone-slash';
                btnToggleMic.classList.add('active-off');
            }
            toggleMute(!isMicOn);
            
            // Gửi tín hiệu về C#
            if(window.chrome && window.chrome.webview) {
                window.chrome.webview.postMessage('mic:' + (isMicOn ? 'on' : 'off'));
            }
        });

        btnToggleCam.addEventListener('click', () => {
            isCamOn = !isCamOn;
            const icon = btnToggleCam.querySelector('i');
            if (isCamOn) {
                icon.className = 'fa-solid fa-video';
                btnToggleCam.classList.remove('active-off');
                if (localStream) localStream.getVideoTracks()[0].enabled = true;
            } else {
                icon.className = 'fa-solid fa-video-slash';
                btnToggleCam.classList.add('active-off');
                if (localStream) localStream.getVideoTracks()[0].enabled = false;
            }
            
            if(window.chrome && window.chrome.webview) {
                window.chrome.webview.postMessage('cam:' + (isCamOn ? 'on' : 'off'));
            }
        });

        btnHangup.addEventListener('click', () => {
            if (window.chrome && window.chrome.webview) {
                window.chrome.webview.postMessage('hangup');
            } else {
                // Test mode trên trình duyệt thường
                updateStatus('closed');
                if(peerConnection) peerConnection.close();
                if(localStream) localStream.getTracks().forEach(track => track.stop());
            }
        });

        // ================= WEBRTC CORE =================

        async function init() {
            try {
                if(localStream) return true;
                updateStatus('new');
                
                localStream = await navigator.mediaDevices.getUserMedia({
                    video: { width: { ideal: 1280 }, height: { ideal: 720 } },
                    audio: { echoCancellation: true, noiseSuppression: true }
                });
                
                localVideo.srcObject = localStream;
                return true;
            } catch (error) {
                console.error("Init Error:", error);
                updateStatus('failed');
                return false;
            }
        }

        function createPeerConnection() {
            if (peerConnection) return peerConnection;
            
            peerConnection = new RTCPeerConnection(config);

            // Add Tracks
            localStream.getTracks().forEach(track => {
                peerConnection.addTrack(track, localStream);
            });

            // On Remote Track
            peerConnection.ontrack = (event) => {
                if (remoteVideo.srcObject !== event.streams[0]) {
                    remoteVideo.srcObject = event.streams[0];
                    console.log("Remote stream received!");
                }
            };

            // On ICE Candidate
            peerConnection.onicecandidate = (event) => {
                if (event.candidate && window.chrome && window.chrome.webview) {
                    const data = [
                        event.candidate.candidate, 
                        event.candidate.sdpMid, 
                        event.candidate.sdpMLineIndex
                    ].join('|');
                    window.chrome.webview.postMessage('candidate:' + data);
                }
            };

            // On Connection State
            peerConnection.onconnectionstatechange = () => {
                console.log("State:", peerConnection.connectionState);
                updateStatus(peerConnection.connectionState);
            };

            return peerConnection;
        }

        function waitForIceGathering(pc) {
            return new Promise((resolve) => {
                if (pc.iceGatheringState === 'complete') {
                    resolve();
                } else {
                    const check = () => {
                        if (pc.iceGatheringState === 'complete') {
                            pc.removeEventListener('icegatheringstatechange', check);
                            resolve();
                        }
                    };
                    pc.addEventListener('icegatheringstatechange', check);
                    setTimeout(resolve, 2000); // Timeout an toàn 2s
                }
            });
        }

        function toggleMute(muted) {
            if(localStream) {
                localStream.getAudioTracks().forEach(t => t.enabled = !muted);
            }
        }

        // ================= PUBLIC FUNCTIONS (CALLED BY C#) =================

        async function createOffer() {
            const inited = await init();
            if(!inited) return;

            const pc = createPeerConnection();
            const offer = await pc.createOffer({offerToReceiveAudio:1, offerToReceiveVideo:1});
            await pc.setLocalDescription(offer);
            
            await waitForIceGathering(pc);
            
            window.chrome.webview.postMessage('offer:' + pc.localDescription.sdp);
        }

        async function setRemoteOffer(sdp) {
            const inited = await init();
            if(!inited) return;

            const pc = createPeerConnection();
            await pc.setRemoteDescription(new RTCSessionDescription({type:'offer', sdp:sdp}));
            
            // Xử lý candidate bị pending
            for(const c of pendingCandidates) {
                await pc.addIceCandidate(c);
            }
            pendingCandidates = [];

            // --- QUAN TRỌNG: TỰ ĐỘNG TẠO ANSWER ---
            await createAnswer();
        }

        async function createAnswer() {
            if(!peerConnection) return;
            
            const answer = await peerConnection.createAnswer();
            await peerConnection.setLocalDescription(answer);
            
            await waitForIceGathering(peerConnection);
            
            window.chrome.webview.postMessage('answer:' + peerConnection.localDescription.sdp);
        }

        async function setRemoteAnswer(sdp) {
            if(!peerConnection) return;
            
            await peerConnection.setRemoteDescription(new RTCSessionDescription({type:'answer', sdp:sdp}));
            
            for(const c of pendingCandidates) {
                await peerConnection.addIceCandidate(c);
            }
            pendingCandidates = [];
        }

        async function addIceCandidate(data) {
            const p = data.split('|');
            const c = new RTCIceCandidate({
                candidate: p[0], 
                sdpMid: p[1], 
                sdpMLineIndex: parseInt(p[2])
            });

            if(peerConnection && peerConnection.remoteDescription) {
                await peerConnection.addIceCandidate(c);
            } else {
                pendingCandidates.push(c);
            }
        }

        console.log('WebRTC Engine Ready');
    </script>
